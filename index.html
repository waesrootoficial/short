let money = parseInt(localStorage.getItem('money')) || 0; let cars = parseInt(localStorage.getItem('cars')) || 0; let moneyPerClick = parseInt(localStorage.getItem('moneyPerClick')) || 1; const moneyDisplay = document.getElementById('money'); const carsDisplay = document.getElementById('cars'); const carClick = document.getElementById('carClick'); const resetButton = document.getElementById('reset'); function updateDisplay() { moneyDisplay.textContent = money; carsDisplay.textContent = cars; } // Click en coche carClick.addEventListener('click', () => { money += moneyPerClick; updateDisplay(); saveGame(); }); // Comprar coches function buyCar(price, increment) { if (money >= price) { money -= price; cars += 1; moneyPerClick += increment; updateDisplay(); saveGame(); } else { alert("No tienes suficiente dinero!"); } } // Guardar progreso function saveGame() { localStorage.setItem('money', money); localStorage.setItem('cars', cars); localStorage.setItem('moneyPerClick', moneyPerClick); } // Resetear juego resetButton.addEventListener('click', () => { if(confirm("¿Estás seguro de resetear?")) { money = 0; cars = 0; moneyPerClick = 1; updateDisplay(); saveGame(); } }); updateDisplay();
